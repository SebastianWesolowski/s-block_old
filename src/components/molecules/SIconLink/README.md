# SIconLink

## Usage

```tsx
import { SIconLink } from "s-block";
import AccessAlarmIcon from "@mui/icons-material/AccessAlarm";
import { SvgLogoSygnet } from "s-brand";
<SIconLink
  path="/"
  label="Home"
  icon={<AccessAlarmIcon sx={{ fontSize: 14 }} />}
/>;

<SIconLink path="/" label="Home" icon={<SvgLogoSygnet />} />;
```
